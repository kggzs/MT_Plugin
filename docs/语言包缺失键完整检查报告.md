## 四、验证方法

### 自动化检查脚本

```python
import os
import re

def check_language_keys():
    base_keys = set()
    lang_files = []
    
    # 读取默认语言包
    with open('strings.mtl', 'r', encoding='utf-8') as f:
        for line in f:
            match = re.match(r'^([^#=]+)=', line)
            if match:
                base_keys.add(match.group(1).strip())
    
    # 检查其他语言包
    for filename in os.listdir('.'):
        if filename.startswith('strings-') and filename.endswith('.mtl'):
            lang_keys = set()
            with open(filename, 'r', encoding='utf-8') as f:
                for line in f:
                    match = re.match(r'^([^#=]+)=', line)
                    if match:
                        lang_keys.add(match.group(1).strip())
            
            missing = base_keys - lang_keys
            if missing:
                print(f"{filename}: 缺失键 {missing}")
            else:
                print(f"{filename}: ✅ 完整")

check_language_keys()
```
